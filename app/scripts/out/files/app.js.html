'use strict';

/**
 *  This modul is used for debugging the bmm Library
 *  
 *  @module bmmDebug
 *  @requires ngResource
 *  @requires ngSanitize
 *  @requires ngRoute
 *  @requires bmmLibApp
 *  @requires ui.sortable
 *  @requires angularTreeview
 */

angular.module('bmmDebug', [
  'ngResource',
  'ngSanitize',
  'ngRoute',
  'bmmLibApp',
  'ui.sortable',
  'angularTreeview'
]).run(['$route', 'bmmApi', function($route, bmmApi)  {
    
    /**
     *  Fires at startup and sets the serverUrl. Will also attempt to connect to server, otherwise redirect to server loginpage
     *  
     *  @event run
     */

    bmmApi.serverUrl('https://'+window.location.hostname+'/bmm/api/web/app_dev.php/');
    
    bmmApi.user().done(function() {
    }).fail(function() {
      bmmApi.loginRedirect();
    });

    $route.reload();

  }])
  .config(function ($routeProvider) {
    $routeProvider
      .when('/', {
        templateUrl: 'views/pages/directives.html',
        controller: 'MainCtrl'
      })
      .when('/bmm_main', {
        templateUrl: 'views/previews/bmm_main.html',
        controller: 'MainCtrl'
      })
      .when('/bmm_sliders', {
        templateUrl: 'views/previews/bmm_sliders.html',
        controller: 'MainCtrl'
      })
      .when('/bmm_api', {
        templateUrl: 'views/previews/bmm_api.html',
        controller: 'ApiCtrl'
      })
      .when('/bmm_player_controller', {
        templateUrl: 'views/previews/bmm_player_controller.html',
      })
      .when('/bmm_player_about', {
        templateUrl: 'views/previews/bmm_player_about.html',
      })
      .otherwise({
        redirectTo: '/'
      });
  });
